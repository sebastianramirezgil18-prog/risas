import java.util.Scanner;
 
public class Main {
    public static void main(String[] args) {
 
        Scanner leer = new Scanner(System.in);
        int tipoCliente;
        String nombre;
        String nombreTipo;
        double compra, totalPagar;
        double dcto1=0, dcto2=0, valorDcto1, valorDcto2 , totalDcto ;
 
        System.out.println("...... TIENDA EN LÍNEA EL MEJOR PRECIO .....");
        System.out.println("Tipo de Cliente");
        System.out.println(" 1 Nuevo");
        System.out.println(" 2 Frecuente");
        System.out.println(" 3 VIP");
        System.out.println();
 
        System.out.print("opcion-->   ");
        tipoCliente = leer.nextInt();
        leer.nextLine(); // <-- Limpiar el buffer
 
        System.out.print("Nombre del cliente: ");
        nombre = leer.nextLine();
 
        System.out.print("Valor de la compra: ");
        compra = leer.nextFloat();
 
        switch (tipoCliente) {
            case 1:
                dcto1 = 5;
                nombreTipo="Nuevo";
                break;
            case 2:
                dcto1 = 10;
                nombreTipo="Frecuente";
                break;
            case 3:
                dcto1 = 15;
                nombreTipo="VIP";
                break;
            default:
                System.out.println("Error al digitar tipo de cliente....");
                nombreTipo="Incorrecto";
                break;
        }

        if (compra >= 10000000) {
            dcto2 = 15;
        } else {
            dcto2 = 5;
        }

        valorDcto1 = compra * dcto1 / 100;
        valorDcto2 = compra * dcto2 / 100;
        totalDcto = valorDcto1 + valorDcto2;
        totalPagar = compra - totalDcto;
         
        System.out.println(".....FACTURA DE COMPRA.....");
        System.out.println("Nombre del cliente: " + nombre);
        System.out.println("Tipo cliente: " + nombreTipo);
        System.out.printf("Valor de la compra inicial: $%,.1f\n",  compra);
        System.out.println("Porcentaje concedido por el tipo cliente: " + dcto1 +"%");
        System.out.println("Valor descuento por tipo cliente:     " + valorDcto1);
        System.out.println("Porcentaje concedido adicional: " + dcto2 + "%");
        System.out.println("Valor descuento adicional:     " +  valorDcto2 );
        System.out.println("Total descuentos: " + totalDcto);
        System.out.println("Total a pagar: " + totalPagar);
    }
}
